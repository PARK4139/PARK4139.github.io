from pypapago import Translator

fileName = r'C:\Users\박 정 훈\Desktop\Daily\tesla.txt'



templist = []

afterTranslate = []

 

with open(fileName, 'r', encoding='utf-8') as f:

    lines = f.readlines()

    for line in lines:

        if '\n' in line:

            line = line.replace('\n', '##jhp')

            templist.append(line)

        else:

            templist.append(line)

        try:

            templist.remove('##jhp')

        except:

            pass

        try:

            for i in range(len(templist)):

                result = Translator().translate(templist[i], source='en', target='ko', verbose=False)

                afterTranslate.append(result)

        except (KeyError):

            pass

    

    



























    

print(lines)

print(afterTranslate)



with open(fileName + '_translated.txt', 'w') as f:

    for i in range(len(afterTranslate)):

        data = afterTranslate[i] + '\n' 

        f.write(data)
