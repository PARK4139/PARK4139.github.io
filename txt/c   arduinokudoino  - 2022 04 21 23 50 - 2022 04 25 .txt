void setup()

{

  // 13(no.) digital pin에는 LED를 연결해두었다.

  // digital 1(no.)pin을 digital output(OUTPUT)모드로 설정한다.

  pinMode(13,OUTPUT); 

}

 

void loop()

{

  digitalWrite(13,HIGH); //13(no.)pin을 HIGH모드로 즉 논리적으로는 ON(true)로 변경

  delay(1000); // 1000ms 기다림 즉 1sec

  digitalWrite(13,LOW); // 13(no.)pin을 LOW모드로 즉 논리적으로는 OFF(false)로 변경

  delay(1000); // 1000ms 기다림 즉 1sec

}
